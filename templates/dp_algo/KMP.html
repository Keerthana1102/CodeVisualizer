void getLPS(string p, int m, int lps[]){
  int index = 0;
  for(int i=1;i<m;){
    if(p[i]==p[index]){
      lps[i] = index+1;
      i++; index++;
    }
    else{
      if(index != 0)   index = lps[index-1];
      else   lps[i] = 0; i++;
  }
}
bool KMP(string s, string p){
  int n = s.length(); int m = p.length(); int lps[m];
  getLPS(p, m, lps);
  int i=0; int j=0;
  while(i < n & j < m){
    if(s[i]==p[j])   i++; j++;
   else {
     if(j!=0)   j = lps[j-1];
     else   i++;
    }
  }
if(j==m) return true;
else return false;
}
