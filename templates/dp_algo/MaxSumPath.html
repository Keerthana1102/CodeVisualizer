int maxSumPath(int A[], int n, int m){
  int DP[n][m];
  for (let i = 0; i < N; i++) {
    for (let j = 0; j < M; j++) {
      if (i == 0 && j == 0)   DP[i][j] = A[i][j];
      else if (i == 0)   DP[i][j] = DP[i][j-1] + A[i][j];
      else if (j == 0)   DP[i][j] = DP[i-1][j] + A[i][j];
      else   DP[i][j] = max(DP[i][j-1], DP[i-1][j]) + A[i][j];
    }
  }
  return DP[n-1][m-1]
}
