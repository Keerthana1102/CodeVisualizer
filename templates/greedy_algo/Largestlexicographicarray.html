(using greedy strategy)

the lexicographically largest array that can be obtained by performing at-most k consecutive swaps



for (let i = 0; i < n - 1 && k > 0; ++i){
      let indexPosition = i;
      for (let j = i + 1; j < n; ++j){
             if (k <= j - i) break;
             if (arr[j] > arr[indexPosition]){
                    indexPosition = j; }
      }
      for (let j = indexPosition; j > i; --j)
             SwapLets(arr, j, j - 1);
      k -= indexPosition - i;}
